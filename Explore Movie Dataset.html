def draw_hist(key_word, space=1):
    revenue_max = movie_data[key_word].max()
    bins = np.arange(0, revenue_max+space,space)
    plt.hist(data=movie_data, x=key_word, bins=bins, color=base_color);
    plt.xlabel(key_word);
    plt.ylabel('count');

def draw_dist(key_word, space=1):
    revenue_max = movie_data[key_word].max()
    bins = np.arange(0, revenue_max+space,space)
    sb.distplot(movie_data[key_word], bins=bins, kde=True,hist_kws={'alpha' : 0.8});

def draw_line(key_word, bins=40):
    x_max = movie_data['revenue'].max()
    x_min = movie_data['revenue'].min()
    space = (x_max-x_min)/(bins+1)
    xbin_edges = np.linspace(x_min, x_max, bins+1)
    xbin_centers = (xbin_edges + space/2)[:-1]
    data_xbins = pd.cut(movie_data['revenue'], xbin_edges, right=False, include_lowest=True)
    y_means = movie_data[key_word].groupby(data_xbins).mean()#
    y_sems = movie_data[key_word].groupby(data_xbins).sem()#
    plt.errorbar(x=xbin_centers, y=y_means, yerr=y_sems);
    plt.ylabel(key_word)
